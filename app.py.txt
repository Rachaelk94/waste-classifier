import streamlit as st
import tensorflow as tf
from tensorflow.keras.preprocessing import image
import numpy as np
from PIL import Image

st.set_page_config(page_title="Waste Classifier", layout="centered")

@st.cache_resource
def load_model():
    return tf.keras.models.load_model("waste_classifier_model.h5")

model = load_model()

class_names = ['cardboard', 'glass', 'metal', 'paper', 'plastic', 'trash']

def preprocess_image(img):
    img = img.resize((224, 224))
    img_array = image.img_to_array(img) / 255.0
    img_array = np.expand_dims(img_array, axis=0)
    return img_array

st.title("♻️ Smart Waste Classifier")
st.write("Upload an image of waste, and the AI will tell you which bin it belongs to.")

uploaded_file = st.file_uploader("Choose an image...", type=["jpg", "jpeg", "png"])

if uploaded_file is not None:
    img = Image.open(uploaded_file).convert("RGB")
    st.image(img, caption="Uploaded Image", use_column_width=True)

    with st.spinner("Analyzing..."):
        processed = preprocess_image(img)
        predictions = model.predict(processed, verbose=0)[0]
        top_idx = np.argmax(predictions)
        top_label = class_names[top_idx]
        top_confidence = predictions[top_idx] * 100

    st.success(f"**Prediction:** {top_label.capitalize()}  ({top_confidence:.1f}% confidence)")
    st.write("**All predictions:**")
    for label, prob in zip(class_names, predictions):
        st.write(f"- {label.capitalize()}: {prob*100:.2f}%")
